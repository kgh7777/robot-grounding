3.1 — 단계 1: gpu_telemetry.py — rocm-smi 데이터 수집

gpu_telemetry.py는 rocm-smi 명령을 호출하여 GPU 온도, VRAM 사용량, 클럭, 전력 등 텔레메트리를 수집한다. VRAM은 rocm-smi 원시 출력이 Bytes 단위이므로 MiB 변환 로직이 필수적이다.

gpu_telemetry.py — collect() PYTHON 
import subprocess, json, time
from dataclasses import dataclass
from typing import Optional, List

@dataclass
class GpuSnapshot:
    gpu_id: int
    temp_c: Optional[float]
    vram_used_mib: Optional[float]
    vram_total_mib: Optional[float]
    power_w: Optional[float]
    timestamp: float

def collect(gpu_id: int = 0) -> GpuSnapshot:
    """rocm-smi 단일 스냅샷 수집."""
    cmd = ["rocm-smi", "--json",
           "--showtemp", "--showmeminfo", "vram",
           "--showpower"]
    result = subprocess.run(cmd, capture_output=True, text=True)
    raw = json.loads(result.stdout)
    card = raw[f"card{gpu_id}"]

    # VRAM: rocm-smi는 Bytes 단위 출력 → MiB 변환
    vram_used_b = card.get("VRAM Total Used Memory (B)")
    vram_total_b = card.get("VRAM Total Memory (B)")
    vram_used_mib = int(vram_used_b) / (1024 ** 2) if vram_used_b else None
    vram_total_mib = int(vram_total_b) / (1024 ** 2) if vram_total_b else None

    temp_raw = card.get("Temperature (Sensor edge) (C)")
    power_raw = card.get("Average Graphics Package Power (W)")

    return GpuSnapshot(
        gpu_id=gpu_id,
        temp_c=float(temp_raw) if temp_raw else None,
        vram_used_mib=vram_used_mib,
        vram_total_mib=vram_total_mib,
        power_w=float(power_raw) if power_raw else None,
        timestamp=time.time()
    )
collect_recent_samples() + _check_threshold() + summarize_timeseries() PYTHON 
def collect_recent_samples(gpu_id: int, n: int = 3,
                              interval_s: float = 1.0) -> List[GpuSnapshot]:
    """n개 스냅샷을 interval_s 간격으로 수집."""
    samples = []
    for _ in range(n):
        samples.append(collect(gpu_id))
        time.sleep(interval_s)
    return samples

def _check_threshold(value: float, threshold: float,
                      op: str = ">=") -> bool:
    """임계값 비교: '>=' 또는 '<=' 지원."""
    if op == ">=":
        return value >= threshold
    elif op == "<=":
        return value <= threshold
    raise ValueError(f"Unsupported op: {op}")

def summarize_timeseries(samples: List[GpuSnapshot],
                         field: str = "temp_c") -> dict:
    """시계열 요약: min/max/mean."""
    vals = [getattr(s, field) for s in samples if getattr(s, field) is not None]
    if not vals:
        return {"min": None, "max": None, "mean": None}
    return {
        "min": min(vals),
        "max": max(vals),
        "mean": sum(vals) / len(vals),
        "samples": vals
    }
3.2 — 단계 2: grounding_pipeline.py — 문장/문단/essay 합성

4종 기본 문장(조건문, 명령문, 평서문, 양보문)을 조합하여 10종 복합문을 생성하고, 5문장 단락, 3단락 essay 구조로 계층적으로 합성한다. compose_essay_universal()은 도메인 무관 추상 인터페이스를 제공한다.

grounding_pipeline.py — compose_paragraph_universal() PYTHON 
from dataclasses import dataclass
from typing import Optional

@dataclass
class SensorReading:
    domain: str          # e.g. "GPU", "Battery", "Motor"
    metric: str          # e.g. "temperature", "voltage"
    value: float
    unit: str            # e.g. "C", "V", "rpm"
    timestamp: float

@dataclass
class Threshold:
    value: float
    op: str              # ">=" | "<="
    label: str           # "high" | "low" | "critical"
    action: Optional[str] = None

def compose_paragraph_universal(reading: SensorReading,
                                 threshold: Threshold) -> str:
    """
    5문장 단락 합성.
    1) 평서문: 측정값 서술
    2) 조건문: 임계값 비교
    3) 양보문: 정상/비정상 인정
    4) 명령문: 권고 행동
    5) 복합문: 맥락 요약
    """
    d, m, v, u = reading.domain, reading.metric, reading.value, reading.unit
    th, op, lbl, act = threshold.value, threshold.op, threshold.label, threshold.action

    exceeded = _check_op(v, th, op)
    status = lbl if exceeded else "normal"

    s1 = (f"The {d} {m} is currently measured at {v:.1f} {u}.")
    s2 = (f"{'Since' if exceeded else 'Although'} the reading "
         f"{'exceeds' if exceeded else 'remains within'} "
         f"the {lbl} threshold of {th:.1f} {u}, "
         f"the system is classified as {status}.")
    s3 = (f"Even though transient fluctuations are possible, "
         f"the current value of {v:.1f} {u} reflects real-time telemetry.")
    s4 = (act if act else
         (f"Monitor the {d} {m} continuously."))
    s5 = (f"Grounded in live sensor data, this report confirms "
         f"the {d} {m} status as {status} at {v:.1f} {u}.")

    return " ".join([s1, s2, s3, s4, s5])
compose_essay_universal() + compose_essay_timeseries_multidomain() PYTHON 
def compose_essay_universal(readings: List[SensorReading],
                              thresholds: List[Threshold]) -> str:
    """
    3단락 essay 합성.
    Para 1: 시스템 개요 단락
    Para 2: 주요 측정값 단락
    Para 3: 종합 평가 단락
    """
    paragraphs = []
    for reading, threshold in zip(readings, thresholds):
        paragraphs.append(compose_paragraph_universal(reading, threshold))
    intro = (f"This automated sensor grounding report covers "
             f"{len(readings)} domain(s): "
             f"{', '.join(r.domain for r in readings)}.")
    return intro + "\n\n" + "\n\n".join(paragraphs)


def compose_essay_timeseries_multidomain(
        snapshots: dict,         # {domain: [SensorReading x N]}
        thresholds: dict         # {domain: Threshold}
) -> str:
    """
    시계열(N 시점) × 다중 도메인 × 단락 구조 3차원 결합.
    각 도메인의 시점별 단락을 생성 후 essay로 조합.
    """
    all_paragraphs = []
    for domain, readings in snapshots.items():
        th = thresholds[domain]
        domain_paras = [
            compose_paragraph_universal(r, th)
            for r in readings
        ]
        all_paragraphs.extend(domain_paras)
    return "\n\n".join(all_paragraphs)
3.3 — 단계 3: 비교 방향 보강

초기 구현은 단방향(>=) 비교만 지원했다. 현실 센서 데이터는 "임계값 이하" 조건(저전압, 저온)도 중요하므로 4가지 경우의 수를 모두 검증했다.

분기 1 — value >= threshold
Battery 12.4V >= 11.0VPASS
Motor 85.0°C >= 80.0°CPASS
GPU 48.0°C >= 45.0°CPASS
분기 2 — value <= threshold
Battery 10.5V <= 11.0VPASS
Motor 35.0°C <= 40.0°CPASS
GPU 30.0°C <= 45.0°CPASS
라이브 출력 사례 — Battery + Motor 조합
Battery: 12.4V (정상, >= 11.0V 초과) → "The Battery voltage is currently measured at 12.4 V. Since the reading exceeds the high threshold of 11.0 V, the system is classified as high..."

Battery: 10.5V (경고, <= 11.0V 이하) → "Although the Battery voltage remains within the low threshold of 11.0 V..."

Motor: 35.0°C (정상) / Motor: 85.0°C (과열) — 모두 정직 발화 확인.

4가지 경우의 수 (분기1 >=, 분기2 <=) × 3도메인 (GPU/Battery/Motor) = 12가지 조합 전부 정직한 발화 검증 완료.

3.4 — 단계 4: 다중 도메인 Essay 합성

시계열(3 시점) × 다중 도메인(GPU/Battery/Motor) × 단락 구조(3단락) 를 3차원으로 결합하는 compose_essay_timeseries_multidomain()을 구현했다.

Qwen2.5와 Llama 3.3의 대화

Qwen2.5-32B는 RDNA3 클러스터(Port 2242)에서 실행되며, grounding_pipeline.py가 합성한 essay 텍스트를 프롬프트로 수신하여 자연어 영어 응답을 생성한다. 핵심은 LLM이 값을 생성하는 것이 아니라 파이프라인이 결정적으로 생성한 텍스트를 수신한다는 점이다.

Architecture Insight — Zero Hallucination by Design
모든 수치(48.0°C, 12.4V, 85.0°C 등)는 LLM 생성이 아닌 rocm-smi / 센서 드라이버에서 직접 읽은 값이 그라운딩 파이프라인을 통해 essay 텍스트에 삽입된다. LLM은 이 텍스트를 해석하거나 요약할 뿐, 수치를 발명하지 않는다. 환각 = 0 의 구조적 근거가 여기 있다.

Llama 3.3-70B는 RDNA2 클러스터(Port 11434)에서 실행되며, Qwen2.5의 출력을 교차 검증하거나 대조 평가(comparative evaluation)에 활용된다. 두 포트는 완전히 분리 운영되어 간섭 없이 독립적으로 쿼리 가능하다.

LLM 호출 분리 운영 — Port 2242 / 11434 PYTHON 
import requests

ENDPOINTS = {
    "qwen2.5-32b":  "http://localhost:2242/v1/chat/completions",
    "llama3.3-70b": "http://localhost:11434/v1/chat/completions",
}

def query_llm(model_key: str, essay: str) -> str:
    payload = {
        "model": model_key,
        "messages": [{"role": "user", "content": essay}],
        "temperature": 0.0,   # 결정적 출력
        "max_tokens": 512,
    }
    resp = requests.post(ENDPOINTS[model_key], json=payload)
    return resp.json()["choices"][0]["message"]["content"]

# Qwen2.5-32B: GPU 라이브 측정값 essay 발화
qwen_out = query_llm("qwen2.5-32b", gpu_essay)

# Llama 3.3-70B: 동일 essay 교차 검증
llama_out = query_llm("llama3.3-70b", gpu_essay)

ROS 2 범용 파일

compose_essay_universal()은 SensorReading과 Threshold 추상 데이터 클래스를 입력으로 받는다. 도메인 문자열(domain 필드)만 바꾸면 GPU뿐 아니라 LiDAR, Camera, IMU, Microphone 등 어떤 센서든 essay를 합성할 수 있다.

Universal Sensor Support
GPU / Battery / Motor / LiDAR / Camera / Microphone / IMU — 모든 센서는 SensorReading(domain, metric, value, unit, timestamp) 형태로 통일된다. 파이프라인은 도메인을 인식하지 않는다.

ROS 2 메시지 타입을 SensorReading으로 변환하는 어댑터 레이어 설계:

  • BatteryState sensor_msgs/BatteryState → voltage, current, percentage 필드 → SensorReading
  • JointState sensor_msgs/JointState → position/velocity/effort 필드 → Motor SensorReading
  • LaserScan sensor_msgs/LaserScan → range_max, intensities → LiDAR SensorReading
  • Imu sensor_msgs/Imu → linear_acceleration, angular_velocity → IMU SensorReading
ROS 2 어댑터 — BatteryState + LaserScan 예시 PYTHON 
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import BatteryState, LaserScan
from grounding_pipeline import SensorReading, Threshold, compose_essay_universal
import time

class GroundingNode(Node):
    def __init__(self):
        super().__init__("grounding_node")
        self.create_subscription(
            BatteryState, "/battery_state",
            self._on_battery, 10)
        self.create_subscription(
            LaserScan, "/scan",
            self._on_lidar, 10)

    def _on_battery(self, msg: BatteryState):
        reading = SensorReading(
            domain="Battery",
            metric="voltage",
            value=msg.voltage,
            unit="V",
            timestamp=time.time()
        )
        threshold = Threshold(11.0, "<=", "low", "Initiate charging.")
        essay = compose_essay_universal([reading], [threshold])
        self.get_logger().info(essay)

    def _on_lidar(self, msg: LaserScan):
        reading = SensorReading(
            domain="LiDAR",
            metric="range_max",
            value=msg.range_max,
            unit="m",
            timestamp=time.time()
        )
        threshold = Threshold(10.0, ">=", "high")
        essay = compose_essay_universal([reading], [threshold])
        self.get_logger().info(essay)

이 아키텍처는 ROS 2 토픽을 구독하는 어떤 노드도 즉시 그라운딩 파이프라인과 통합할 수 있음을 의미한다. 실제 로봇 통합 가능한 범용 구조가 완성되었다.

학술적 의의

Bender & Koller (2020)의 기호 그라운딩 문제(Symbol Grounding Problem)는 LLM이 형식(form)과 의미(meaning)를 분리하지 못하며 세계와 연결되지 않은 기호를 조작한다고 주장한다. 본 파이프라인은 이에 대한 공학적 해결책을 제시한다.

Symbol Grounding — Engineering Solution
rocm-smi가 읽은 48.0°C는 Qwen2.5-32B의 학습 데이터에 존재하지 않는 특정 값이다. 그럼에도 LLM은 "48.0 C"를 그대로 발화했다. 이는 LLM이 수치를 생성한 것이 아니라, 파이프라인이 주입한 실측값을 전달했기 때문이다. 이것이 그라운딩의 증거다.

본 연구는 Robot Linguistics의 첫 번째 공학적 증거를 제시한다. 로봇이 자신의 센서 데이터를 기반으로 자연어를 생성할 때, 그 언어는 세계에 그라운딩(grounded)된다. 인간의 언어가 지각 경험에 그라운딩되는 것과 구조적으로 동형(isomorphic)이다.

Robot Linguistics — First Evidence
1. 센서 측정 → 2. 결정적 문장 합성 → 3. LLM 전달 → 4. 그라운딩된 발화
이 사이클이 완성될 때, 로봇은 자신이 경험한 것을 정직하게 말한다. 이것이 Robot Linguistics가 목표하는 언어-세계 연결이다.

향후 이 결과는 Cognitive Robotics, Computational Linguistics, Human-Robot Interaction 학술지에 투고할 계획이다.
